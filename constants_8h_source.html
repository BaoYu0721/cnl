26  template<typename Rep, int Exponent>
27  constexpr auto pi(int const max_iterations) {
28  using fp = fixed_point<Rep, Exponent>;
29  constexpr auto four = fixed_point<Rep, 3 - digits<Rep>::value>{4.};
30 
31  auto previous = fp{3.};
32  for(auto n = 2; n != (max_iterations << 1); n += 4) {
33  auto const addend = four / ((n+0L) * (n+1L) * (n+2L));
34  auto const subtrahend = four / ((n+2L) * (n+3L) * (n+4L));
35  auto next = fp{previous + addend - subtrahend};
36  if (next == previous) {
37  return next;
38  }
39  previous = next;
40  }
41  return previous;
42  }

49  template<typename Rep, int Exponent>
50  constexpr auto e(int const max_iterations) {
51  using fp = fixed_point<Rep, Exponent>;
52  constexpr auto one = fixed_point<Rep, 1 - digits<Rep>::value>{1.};
53 
54  auto previous = fp{2.};
55  auto factor = 2;
56  for (auto n = 2; n != max_iterations; ++ n, factor *= n) {
57  auto const addend = one / factor;
58  auto next = fp{previous + addend};
59  if (next == previous) {
60  return next;
61  }
62  previous = next;
63  }
64  return previous;
65  }

72  // Given two alternative ways to generate a constant value:
73  // tries to choose the best and returns the result.
74  template<typename Float, typename Rep, int Exponent>
75  constexpr auto constant_with_fallback(Float constant, fixed_point<Rep, Exponent>(*procedure)()) {
76  using fp = fixed_point<Rep, Exponent>;
77 
78  auto const required_integer_digits = used_bits(static_cast<int>(constant));
79 
80  constexpr auto fixed_fractional_digits = fractional_digits<fp>::value;
81  // expect: integer_digits<fp>::value >= required_integer_digits
82 
83  constexpr auto float_digits = std::numeric_limits<Float>::digits;
84  auto const float_fractional_digits = float_digits - required_integer_digits;
85 
86  return (float_fractional_digits >= fixed_fractional_digits)
87  ? fp{constant}
88  : procedure();
89  }
90  }
