54  namespace _impl {
55  template<class Rep>
56  constexpr Rep sqrt_solve3(
57  Rep n,
58  Rep bit,
59  Rep result)
60  {
61  return (bit!=Rep{0})
62  ? (n>=result+bit)
63  ? sqrt_solve3<Rep>(
64  static_cast<Rep>(n-(result+bit)),
65  static_cast<Rep>(bit >> 2),
66  static_cast<Rep>((result >> 1)+bit))
67  : sqrt_solve3<Rep>(
68  n,
69  static_cast<Rep>(bit >> 2),
70  static_cast<Rep>(result >> 1))
71  : result;
72  }

74  template<int Exponent>
75  struct sqrt_solve1 {
76  template<class Rep>
77  constexpr Rep operator()(Rep n) const
78  {
79  using widened_rep = _impl::set_width_t<Rep, _impl::width<Rep>::value*2>;
80  return static_cast<Rep>(sqrt_solve3<widened_rep>(
81  _impl::fixed_width_scale<-Exponent>(static_cast<widened_rep>(n)),
82  widened_rep((widened_rep{1}<<((countr_used(n)+1-Exponent)&~1))>>2),
83  widened_rep{0}));
84  }
85  };
86  }

103  template<typename Rep, int Exponent, int Radix>
104  constexpr auto
105  sqrt(fixed_point<Rep, Exponent, Radix> const& x)
106  -> fixed_point <Rep, Exponent, Radix>
107  {
108  using type = fixed_point<Rep, Exponent, Radix>;
109  return _impl::to_rep(x)<0
110  ? _impl::unreachable<type>("negative value passed to cnl::sqrt")
111  : type{_impl::from_rep<type>(_impl::sqrt_solve1<Exponent>{}(unwrap(x)))};
112  }
