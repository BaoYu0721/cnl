134  // performs zero-degreen binary operations between fixed_point types with the different exponents
135  template<typename Operator, typename LhsRep, int LhsExponent, typename RhsRep, int RhsExponent, int Radix>
136  struct binary_operator<Operator, fixed_point<LhsRep, LhsExponent, Radix>, fixed_point<RhsRep, RhsExponent, Radix>,
137  enable_if_t<is_zero_degree<Operator>::value, typename Operator::is_not_comparison>> {
138  private:
139  static constexpr int _common_exponent = min(LhsExponent, RhsExponent);
140  static constexpr int _lhs_left_shift = LhsExponent-_common_exponent;
141  static constexpr int _rhs_left_shift = RhsExponent-_common_exponent;
142  public:
143  constexpr auto operator()(
144  fixed_point<LhsRep, LhsExponent, Radix> const& lhs, fixed_point<RhsRep, RhsExponent, Radix> const& rhs) const
145  -> decltype(Operator{}(
146  from_rep<fixed_point<LhsRep, _common_exponent, Radix>>{}(
147  _impl::shift<_lhs_left_shift>(to_rep(lhs))),
148  from_rep<fixed_point<RhsRep, _common_exponent, Radix>>{}(
149  _impl::shift<_rhs_left_shift>(to_rep(rhs)))))
150  {
151  return Operator{}(
152  from_rep<fixed_point<LhsRep, _common_exponent, Radix>>{}(
153  _impl::shift<_lhs_left_shift>(to_rep(lhs))),
154  from_rep<fixed_point<RhsRep, _common_exponent, Radix>>{}(
155  _impl::shift<_rhs_left_shift>(to_rep(rhs))));
156  }
157  };
