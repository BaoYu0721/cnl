83  return fp{multiply(xf, (coeffs::a1+fp{multiply(xf, (coeffs::a2+fp{multiply(xf, (coeffs::a3+fp{multiply(xf, (coeffs::a4

100  -> _impl::enable_if_t<(Exponent<0), make_largest_ufraction<fixed_point<Rep, Exponent>>> {

122  template<class Intermediate, typename Rep, int Exponent>
123  constexpr typename Intermediate::rep
124  exp2(fixed_point<Rep, Exponent> const& x, Rep const& floored) {
125  return floored <= Exponent
126  ? typename Intermediate::rep{1}//return immediately if the shift would result in all bits being shifted out
127  //Do the shifts manually. Once the branch with shift operators is merged, could use those
128  : (to_rep(exp2m1_0to1<Rep, Exponent>(fractional(x, floored)))//Calculate the exponent of the fractional part
129  >> (-Intermediate::exponent + Exponent - floored))//shift it to the right place
130  + (Rep { 1 } << (floored - Exponent)); //The constant term must be one, to make integer powers correct
131  }
132  }

143  template<class Rep, int Exponent>
144  constexpr fixed_point<Rep, Exponent> exp2(fixed_point<Rep, Exponent> x) {
145  using out_type = fixed_point<Rep, Exponent>;
146  // The input type
147  using im = _impl::fp::make_largest_ufraction<out_type>;
148 
149  //Calculate the final result by shifting the fractional part around.
150  //Remember to add the 1 which is left out to get 1 bit more resolution
151  return from_rep<out_type>{}(_impl::fp::exp2<im>(x, static_cast<Rep>(floor(x))));
152  }
