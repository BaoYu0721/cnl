55  template<class Lhs, class Rhs>
56  struct params {
57  using lhs_type = typename fixed_point_type<Lhs>::type;
58  using rhs_type = typename fixed_point_type<Rhs>::type;
59  using lhs_rep = typename lhs_type::rep;
60  using rhs_rep = typename rhs_type::rep;
61 
62  // Exponent
63  static constexpr int rep_exponent = lhs_type::exponent + rhs_type::exponent;
64 
65  // Rep
66  using rep_op_result = _impl::op_result<_impl::multiply_op, lhs_rep, rhs_rep>;
67  static constexpr int sum_digits = digits<lhs_type>::value + digits<rhs_type>::value;
68  static constexpr bool is_signed =
69  numeric_limits<lhs_rep>::is_signed || numeric_limits<rhs_rep>::is_signed;
70  using prewidened_result_rep = _impl::make_signed_t<rep_op_result, is_signed>;
71  using result_rep = set_digits_t<prewidened_result_rep, sum_digits>;
72 
73  // If the 'natural' result of the rep op is wide enough, stick with it.
74  // This ensures that auto-widening rep types (e.g. elastic_integer) don't get widened twice
75  // but types that need a little help (e.g. built-ins) get widened going into the op.
76  using rep_type = typename std::conditional<
77  digits<prewidened_result_rep>::value >= sum_digits,
78  lhs_rep, result_rep>::type;
79 
80  using result_type = fixed_point<result_rep, rep_exponent>;
81  using intermediate_lhs = fixed_point<rep_type, lhs_type::exponent>;
82  using intermediate_rhs = Rhs;
83  };
