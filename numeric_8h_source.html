138  namespace _numeric_impl {
139  template<class Integer>
140  constexpr int used_bits_positive(Integer const& value, int mask_bits = cnl::digits<Integer>::value/2)
141  {
142  static_assert(cnl::numeric_limits<Integer>::is_integer,
143  "Integer parameter of used_bits_positive() must be a fundamental integer.");
144 
145  return (value>=(Integer{1} << mask_bits))
146  ? mask_bits+used_bits_positive(value/(Integer{1} << mask_bits), mask_bits)
147  : (mask_bits>1)
148  ? used_bits_positive(value, mask_bits/2)
149  : 1;
150  }
151  }

163  template<>
164  struct used_bits<true> {
165  template<class Integer>
166  constexpr int operator()(Integer const& value) const
167  {
168  static_assert(cnl::numeric_limits<Integer>::is_integer,
169  "Integer parameter of used_bits()() must be a fundamental integer.");
170 
171  // Most negative number is not exploited;
172  // thus negating the result or subtracting it from something else
173  // will less likely result in overflow.
174  return (value>0)
175  ? used_bits_positive(value)
176  : (value==0)
177  ? 0
178  : used_bits()(Integer(-1)-value);
179  }
180  };
181  }
