284  namespace _numeric_impl {
285  template<class Integer, bool IsSigned>
286  struct trailing_bits {
287  constexpr int operator()(Integer const& integer) const noexcept
288  {
289  return countr_zero(integer);
290  }
291  };
292 
293  template<class Integer>
294  struct trailing_bits<Integer, true> {
295  constexpr int operator()(Integer const& integer) const noexcept
296  {
297  using unsigned_type = remove_signedness_t<Integer>;
298  return countr_zero(static_cast<unsigned_type>(integer));
299  }
300  };
301  }
302 
303  // count of the right redundant trailing bits
304  template<class Integer>
305  constexpr int trailing_bits(Integer const& value)
306  {
307  return value ? _numeric_impl::trailing_bits<Integer, is_signed<Integer>::value>()(value) : 0;
308  }
