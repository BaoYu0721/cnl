274  namespace _numeric_impl {
275  template<class Integer, bool IsSigned>
276  struct trailing_bits {
277  constexpr int operator()(Integer const& integer) const noexcept
278  {
279  return countr_zero(integer);
280  }
281  };
282 
283  template<class Integer>
284  struct trailing_bits<Integer, true> {
285  constexpr int operator()(Integer const& integer) const noexcept
286  {
287  using unsigned_type = make_unsigned_t<Integer>;
288  return countr_zero(static_cast<unsigned_type>(integer));
289  }
290  };
291  }
292 
293  // count of the right redundant trailing bits
294  template<class Integer>
295  constexpr int trailing_bits(Integer const& value)
296  {
297  return value ? _numeric_impl::trailing_bits<Integer, is_signed<Integer>::value>()(value) : 0;
298  }
